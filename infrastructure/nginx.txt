 server{
     server_name dev.nejlepsi-skolky.cz www.dev.svatba.vip 62.171.159.195; # do not use http:
     # Set max body size for POST requests
     client_max_body_size 10M;
     location /{
         proxy_pass "http://127.0.0.1:30001";
         # Add CORS headers
         add_header "Access-Control-Allow-Origin" "*";
         add_header "Access-Control-Allow-Methods" "GET, POST, OPTIONS, PUT, DELETE";
         add_header "Access-Control-Allow-Headers" "Content-Type, Authorization";
         add_header "Access-Control-Allow-Credentials" "true";
     }
     # Handle the POST endpoint
     location /api/ {
         limit_req zone=requestlimit burst=10 nodelay;
         proxy_pass http://127.0.0.1:4002; #  backend service port
         proxy_set_header Host $host;
         proxy_set_header X-Real-IP $remote_addr;
         proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
         proxy_set_header X-Forwarded-Proto $scheme;
         # Enable proxy buffering for large POST requests (optional)
         proxy_buffering off;
     }

     # Logging
     access_log /var/log/nginx/accessMalickovBE.log;
     error_log /var/log/nginx/errorMalickovBE.log;

}
